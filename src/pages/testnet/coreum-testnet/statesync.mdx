### Setup
This is the settings we use , you dont have to change anything

- snapshot-interval= 1000
- snapshot-keep-recent = 5

### Intruction

```
systemctl stop cored
cored tendermint unsafe-reset-all --home $HOME/.core/coreum-testnet-1 --keep-addr-book

STATE_SYNC_RPC="https://rpc.coreum-t.indonode.net:443"

LATEST_HEIGHT=$(curl -s $STATE_SYNC_RPC/block | jq -r .result.block.header.height)
SYNC_BLOCK_HEIGHT=$(($LATEST_HEIGHT - 1000))
SYNC_BLOCK_HASH=$(curl -s "$STATE_SYNC_RPC/block?height=$SYNC_BLOCK_HEIGHT" | jq -r .result.block_id.hash)

PEERS=bacd8143d11f0713d35c59e11ed3fd6f2945bc60@rpc.coreum-t.indonode.net:31656 
sed -i.bak -e "s/^persistent_peers *=.*/persistent_peers = \"$PEERS\"/" $HOME/.core/coreum-testnet-1/config/config.toml

sed -i.bak -e "s|^enable *=.*|enable = true|" $HOME/.core/coreum-testnet-1/config/config.toml
sed -i.bak -e "s|^rpc_servers *=.*|rpc_servers = \"$STATE_SYNC_RPC,$STATE_SYNC_RPC\"|" \
  $HOME/.core/coreum-testnet-1/config/config.toml
sed -i.bak -e "s|^trust_height *=.*|trust_height = $SYNC_BLOCK_HEIGHT|" \
  $HOME/.core/coreum-testnet-1/config/config.toml
sed -i.bak -e "s|^trust_hash *=.*|trust_hash = \"$SYNC_BLOCK_HASH\"|" \
  $HOME/.core/coreum-testnet-1/config/config.toml

systemctl restart cored && journalctl -fu cored -o cat
```
