### Setup
This is the settings we use , you dont have to change anything

- snapshot-interval= 1000
- snapshot-keep-recent = 5

### Intruction

```
systemctl stop pointd
pointd tendermint unsafe-reset-all --home $HOME/.pointd --keep-addr-book

STATE_SYNC_RPC="https://rpc.point.indonode.net:443"

LATEST_HEIGHT=$(curl -s $STATE_SYNC_RPC/block | jq -r .result.block.header.height)
SYNC_BLOCK_HEIGHT=$(($LATEST_HEIGHT - 1000))
SYNC_BLOCK_HASH=$(curl -s "$STATE_SYNC_RPC/block?height=$SYNC_BLOCK_HEIGHT" | jq -r .result.block_id.hash)

PEERS=9c7902cf292e1dbb79b2a0c3ed18f9e6f826483f@rpc.point.indonode.net:21656 
sed -i.bak -e "s/^persistent_peers *=.*/persistent_peers = \"$PEERS\"/" $HOME/.pointd/config/config.toml

sed -i.bak -e "s|^enable *=.*|enable = true|" $HOME/.pointd/config/config.toml
sed -i.bak -e "s|^rpc_servers *=.*|rpc_servers = \"$STATE_SYNC_RPC,$STATE_SYNC_RPC\"|" \
  $HOME/.pointd/config/config.toml
sed -i.bak -e "s|^trust_height *=.*|trust_height = $SYNC_BLOCK_HEIGHT|" \
  $HOME/.pointd/config/config.toml
sed -i.bak -e "s|^trust_hash *=.*|trust_hash = \"$SYNC_BLOCK_HASH\"|" \
  $HOME/.pointd/config/config.toml

systemctl restart pointd && journalctl -fu pointd -o cat
```
